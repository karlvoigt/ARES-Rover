#Add the linker script to use to the linker flags

SET(STARTUP_SRC startup/startup_${PLATFORM_MCU}.s)
ENABLE_LANGUAGE(ASM)
SET_PROPERTY(SOURCE ${STARTUP_SRC} PROPERTY LANGUAGE ASM)

INCLUDE_DIRECTORIES(
    inc
    Drivers/CMSIS/Device/ST/STM32L1xx/Include
    Drivers/CMSIS/Include
    Drivers/STM32L1xx_HAL_Driver/Inc
    Drivers/STM32L1xx_HAL_Driver/Inc/Legacy
)

##Export the 'inc' directory globally
EXPORT_GLOBAL_INCLUDE_DIRECTORIES(
    inc
    Drivers/STM32L1xx_HAL_Driver/Inc
    Drivers/CMSIS/Device/ST/STM32L1xx/Include
    Drivers/CMSIS/Include
)

#Please note we include 'inc' here since for complicated cmake reasons directories exported
#with 'GLOBAL_INCLUDE_DIRECTORIES'from chip directories are not exported to the platform directory
EXPORT_PLATFORM_INCLUDE_DIRECTORIES(
    inc
    Drivers/STM32L1xx_HAL_Driver/Inc
    Drivers/CMSIS/Device/ST/STM32L1xx/Include
    Drivers/CMSIS/Include
)

#An object library with name '${CHIP_LIBRARY_NAME}' MUST be generated by th CMakeLists.txt file for every chip
ADD_LIBRARY (${CHIP_LIBRARY_NAME} OBJECT
    ${STARTUP_SRC}
    #stm32l0xx_hal_msp.c
    system_stm32l1xx.c
    Drivers/STM32L1xx_HAL_Driver/Src/stm32l1xx_hal.c
    Drivers/STM32L1xx_HAL_Driver/Src/stm32l1xx_hal_cortex.c
    Drivers/STM32L1xx_HAL_Driver/Src/stm32l1xx_hal_gpio.c
    Drivers/STM32L1xx_HAL_Driver/Src/stm32l1xx_hal_i2c.c
    Drivers/STM32L1xx_HAL_Driver/Src/stm32l1xx_hal_iwdg.c
    Drivers/STM32L1xx_HAL_Driver/Src/stm32l1xx_hal_pwr.c
    Drivers/STM32L1xx_HAL_Driver/Src/stm32l1xx_hal_pwr_ex.c
    Drivers/STM32L1xx_HAL_Driver/Src/stm32l1xx_hal_rcc.c
    Drivers/STM32L1xx_HAL_Driver/Src/stm32l1xx_hal_rcc_ex.c
    Drivers/STM32L1xx_HAL_Driver/Src/stm32l1xx_hal_rtc.c
    Drivers/STM32L1xx_HAL_Driver/Src/stm32l1xx_hal_rtc_ex.c
    Drivers/STM32L1xx_HAL_Driver/Src/stm32l1xx_hal_spi.c
    Drivers/STM32L1xx_HAL_Driver/Src/stm32l1xx_hal_spi_ex.c
    Drivers/STM32L1xx_HAL_Driver/Src/stm32l1xx_hal_tim.c
    Drivers/STM32L1xx_HAL_Driver/Src/stm32l1xx_hal_tim_ex.c
    Drivers/STM32L1xx_HAL_Driver/Src/stm32l1xx_hal_uart.c
    Drivers/STM32L1xx_HAL_Driver/Src/stm32l1xx_ll_gpio.c
)
